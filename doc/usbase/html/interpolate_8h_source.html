21 template<class X, class Y> QPair<int,int> bracket(const QVector<X> increasing, Y x) {
22  if (increasing.isEmpty()) {
23  QString msg{"Cannot bracket value '%1' in empty vector"};
24  throw Exception(msg.arg(x));
25  }
26 
27  auto p1 = increasing.constBegin();
28  if (x <= *p1 || increasing.size() == 1) return qMakePair(0,0);
29 
30  int end = increasing.size() - 1;
31  if (x >= increasing.at(end)) return qMakePair(end,end);
32 
33  auto p2 = p1; ++p2;
34  int i = 0;
35  while (x > *p2) { ++p1; ++p2; ++i; }
36  return qMakePair(i, i+1);
37 }

44 double interpolate(QPair<X,Y> p1, QPair<X,Y> p2, Z x) {
45  if (p2.first == p1.first) {
46  QString msg{"Cannot interpolate between two points with same x: (%1,%2) and (%3,%4)"};
47  throw Exception(msg.arg(p1.first).arg(p1.second).arg(p2.first).arg(p2.second));
48  }
49  return double(x - p1.first)/(p2.first - p1.first)*(p2.second - p1.second) + p1.second;
50 
51 }

58 template<class X, class Y>
59 double interpolate(const QMap<X,Y> &xy, X x) {
60  if (xy.isEmpty()) {
61  QString msg{"Cannot interpolate value '%1' from empty (x,y)-table"};
62  throw Exception(msg.arg(x));
63  }
64  Y lastY = xy[0];
65  auto low = xy.begin();
66  for (auto high = xy.constBegin(); high != xy.constEnd(); ++high) {
67  if (x < high.key()) {
68  if (high.key() == low.key()) return low.value();
69  return interpolate(qMakePair(low.key(), low.value()), qMakePair(high.key(), high.value()), x);
70  }
71  lastY = high.value();
72  low = high;
73  }
74  return lastY;
75 }

78 template<class X> double interpolate(const DataGrid &matrix, X rowValue, X colValue) {
79  if (matrix.rowNumber() < 2 || matrix.columnNumber() < 3) {
80  QString msg{"Matrix (%1x%2) must at least be 2x2 to perform interpolation"};
81  throw Exception(msg.arg(matrix.rowNumber()).arg(matrix.columnNumber()-1));
82  }
83  X x{rowValue}, y{colValue};
84  QStringList xs = matrix.rowNames(), ys = matrix.columnNames();
85  ys[0] = ys[1]; // remove dummy column name
86  QVector<double> xIncreasing, yIncreasing;
87  for (QString s : xs) {
88  bool ok;
89  xIncreasing << s.toDouble(&ok);
90  if (!ok) {
91  QString msg{"Cannot interpolate. This is not a number: '%1'"};
92  throw Exception(msg.arg(s));
93  }
94  }
95  for (QString s : ys) {
96  bool ok;
97  yIncreasing << s.toDouble(&ok);
98  if (!ok) {
99  QString msg{"Cannot interpolate. This is not a number: '%1'"};
100  throw Exception(msg.arg(s));
101  }
102  }
103 
104  QPair<int,int> xb = bracket(xIncreasing, x),
105  yb = bracket(yIncreasing, y);
106  double x1 = xIncreasing.at(xb.first),
107  x2 = xIncreasing.at(xb.second),
108  y1 = yIncreasing.at(yb.first),
109  y2 = yIncreasing.at(yb.second),
110  z11 = matrix.cell<double>(xb.first, yb.first),
111  z21 = matrix.cell<double>(xb.second, yb.first),
112  z12 = matrix.cell<double>(xb.first, yb.second),
113  z22 = matrix.cell<double>(xb.second, yb.second);
114  double r1 = interpolate(qMakePair(x1, z11), qMakePair(x2, z21), x),
115  r2 = interpolate(qMakePair(x1, z12), qMakePair(x2, z22), x);
116  return interpolate(qMakePair(y1, r1), qMakePair(y2, r2), y);
117 }
