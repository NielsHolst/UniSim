309 template <class T> QList<T> NamedObject::seekSiblings(QString name, int *ixPreceding) {
310  QList<T> siblingsAndMe;
311  if (!parent())
312  return siblingsAndMe;
313  NamedObject *parent = seekParent<NamedObject*>("*");
314  siblingsAndMe = parent->seekChildren<T>(name);

429 template <class T> QList<T> NamedObject::filterByName(QString name, const QList<QObject*> &candidates) {
430  QList<T> result;
431  QStringList names = Identifier(name).key().split("/");
432  if (!names.isEmpty() && names.first().isEmpty())
433  names.removeFirst();
434  if (!names.isEmpty() && names.last().isEmpty())
435  names.removeLast();
436  if (names.isEmpty())
437  return result;
438 
439  for (auto candidate : candidates) {
440 // for (int ca = 0; ca < candidates.size(); ++ca) {
441  int i = names.size()-1;
442  QObject *p{candidate};
443 // QObject *candidate, *p;
444 // candidate = p = candidates.at(ca);
445  while (i > 0 && p->parent() && (p->objectName() == names[i] || names[i] == "*")) {
446  --i;
447  p = p->parent();
448  }
449  if (names[i].isEmpty())
450  throw Exception("Composite component name contains an empty name: " + name, candidate);
451  bool isT = dynamic_cast<T>(candidate) != 0;
452  bool pathOk = i == 0 && (p->objectName() == names[0] || names[0] == "*");
453  if (isT && pathOk) {
454  result << dynamic_cast<T>(candidate);
455  }
456  }
457  return result;
458 }
