141 template<class X, class Y>
142 double interpolate(const QMap<X,Y> &xy, X x) {
143  if (xy.size() == 0) return 0;
144  Y lastY = xy[0];
145  auto low = xy.begin();
146  for (QMap<X,Y>::const_iterator high = xy.begin();
147  high != xy.end();
148  ++high)
149  {
150  if (x < high.key()) {
151  if (high.key() == low.key()) return low.value();
152  return interpolate(qMakePair(low.key(), low.value()), qMakePair(high.key(), high.value()), x);
153  }
154  lastY = high.value();
155  low = high;
156  }
157  return lastY;
158 }

210 template<class T, class U>
211 QMap<T,U> decodeList(QString s, QObject *context) {
212  QMap<T,U> result;
213  QStringList pairs = decodeList(s, context);
214  for (int i = 0; i < pairs.size(); ++i) {
215  QStringList pair = decodeList(pairs[i]);
216  if (pair.size() != 2)
217  throw Exception("(name value) pair expected, got: \"" + pairs[i] + "\"", context);
218  T key = stringToValue<T>(pair[0], context);
219  U value = stringToValue<U>(pair[1], context);
220  if (result.contains(key))
221  throw Exception("Name must be unique in list, found \"" + pair[0] + "\" twice");
222  result[key] = value;
223  }
224  return result;

227 template <class T>
228 QList< QPair<QString, T> > decodeNameValueList(QString nameValueList, QObject *concerning) {
229  QList< QPair<QString, T> > result;
230  QString s = nameValueList.simplified();
231  s = s.trimmed();
232  if (s.size() == 0) throw Exception("Name-value list is empty", concerning);
233  if (s.left(1) != "(") throw Exception("Name-value list must begin with '(': " + s, concerning);
234 
235  QStringList parts = s.split("(");
236  for (int i = 0; i < 2; ++i) {
237  if (parts[i].size() > 0 && parts[i].left(1) != " ")
238  throw UniSim::Exception("Name-value list must begin with two left parentheses: " + s, concerning);
239  }
240 
241  if (parts.size() < 3)
242  throw UniSim::Exception("Name-value list is incomplete: " + s, concerning);
243 
244  for (int i = 2; i< parts.size(); ++i) {
245  QString part = parts[i].trimmed();
246  chopRightParenthesis(part, concerning);
247  if (i == parts.size() - 1)
248  chopRightParenthesis(part, concerning);
249 
250  QStringList strPair = part.split(" ");
251  bool ok = strPair.size() == 2;
252  if (ok) {
253  QPair<QString, T> pair;
254  pair.first = strPair[0];
255  pair.second = stringToValue<T>(strPair[1], concerning);
256  result.append(pair);
257  }
258 
259  if (!ok)
260  throw UniSim::Exception("Name-value list must contain (name value) pairs: " + s, concerning);
261  }
262  return result;
263 }
